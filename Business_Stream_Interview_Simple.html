Q1: Rolling Consumption Anomaly Detection

Scenario: Identify customers whose 3-month consumption exceeds 2.5 standard deviations above their 12-month rolling average.

Skills Tested: Window functions, statistical analysis, time-series

WITH MonthlyConsumption AS (
  SELECT CustomerID, DATE_TRUNC('month', ReadingDate) AS Month, 
    SUM(Consumption) AS MonthlyTotal
  FROM MeterReadings 
  GROUP BY CustomerID, DATE_TRUNC('month', ReadingDate)
),
RollingStats AS (
  SELECT CustomerID, Month, MonthlyTotal,
    AVG(MonthlyTotal) OVER (PARTITION BY CustomerID ORDER BY Month 
      ROWS BETWEEN 12 PRECEDING AND 1 PRECEDING) AS RollingAvg,
    STDDEV(MonthlyTotal) OVER (PARTITION BY CustomerID ORDER BY Month 
      ROWS BETWEEN 12 PRECEDING AND 1 PRECEDING) AS RollingStdDev
  FROM MonthlyConsumption
)
SELECT CustomerID, Month, MonthlyTotal, RollingAvg, RollingStdDev,
  (MonthlyTotal - RollingAvg) / NULLIF(RollingStdDev, 0) AS ZScore
FROM RollingStats
WHERE ABS((MonthlyTotal - RollingAvg) / NULLIF(RollingStdDev, 0)) > 2.5;

Q2: Recursive CTE for Customer Hierarchy

Build a recursive query to traverse organizational customer hierarchies for consolidated billing.

WITH RECURSIVE CustomerHierarchy AS (
  SELECT CustomerID, ParentCustomerID, CustomerName, 0 AS Level,
    CAST(CustomerName AS VARCHAR(500)) AS Path
  FROM Customers WHERE ParentCustomerID IS NULL
  UNION ALL
  SELECT c.CustomerID, c.ParentCustomerID, c.CustomerName, ch.Level + 1,
    ch.Path + ' > ' + c.CustomerName
  FROM Customers c
  INNER JOIN CustomerHierarchy ch ON c.ParentCustomerID = ch.CustomerID
)
SELECT * FROM CustomerHierarchy ORDER BY Path;

Q5: Customer Segmentation Using SQL

Segment customers into quartiles based on consumption patterns for targeted interventions.

WITH CustomerMetrics AS (
  SELECT CustomerID, AVG(Consumption) AS AvgConsumption,
    PERCENTILE_CONT(0.5) WITHIN GROUP (ORDER BY Consumption) AS MedianConsumption
  FROM MeterReadings GROUP BY CustomerID
),
Segmented AS (
  SELECT *, NTILE(4) OVER (ORDER BY AvgConsumption) AS ConsumptionQuartile,
    CASE WHEN AvgConsumption > MedianConsumption * 1.5 THEN 'High Variability'
         WHEN AvgConsumption < MedianConsumption * 0.5 THEN 'Low Usage'
         ELSE 'Standard' END AS UsagePattern
  FROM CustomerMetrics
)
SELECT * FROM Segmented;

Q6: Leakage Detection Pattern Query

Identify potential leakage through continuous low-flow patterns overnight.

WITH NightReadings AS (
  SELECT CustomerID, ReadingDate, Consumption,
    LAG(Consumption) OVER (PARTITION BY CustomerID ORDER BY ReadingDate) AS PrevConsumption
  FROM MeterReadings WHERE DATEPART(hour, ReadingDate) BETWEEN 0 AND 5
),
LeakageCandidates AS (
  SELECT CustomerID, COUNT(*) AS NightReadingsCount,
    AVG(Consumption - PrevConsumption) AS AvgNightFlow
  FROM NightReadings WHERE Consumption > PrevConsumption
  GROUP BY CustomerID
  HAVING COUNT(*) > 20 AND AVG(Consumption - PrevConsumption) > 0.1
)
SELECT * FROM LeakageCandidates;

Q2: Data Cleaning Pipeline

Handle messy utility meter data with missing values and outliers.

def clean_meter_data(df):
    # Remove duplicates
    df = df.drop_duplicates(subset=['meter_id', 'reading_date'])
    
    # Handle negative consumption
    df.loc[df['consumption'] < 0, 'consumption'] = np.nan
    
    # Outlier detection using IQR
    Q1 = df['consumption'].quantile(0.25)
    Q3 = df['consumption'].quantile(0.75)
    IQR = Q3 - Q1
    outlier_mask = (df['consumption'] < Q1 - 1.5*IQR) | (df['consumption'] > Q3 + 1.5*IQR)
    df.loc[outlier_mask, 'consumption'] = np.nan
    
    return df

@dataclass
class ETLConfig:

Q10: Custom Anomaly Detection

Implement statistical anomaly detection algorithm.

class ZScoreAnomalyDetector:
    def __init__(self, threshold=3):
        self.threshold = threshold
        self.mean = None
        self.std = None
    
    def fit(self, data):
        self.mean = np.mean(data)
        self.std = np.std(data)
        return self
    
    def predict(self, data):
        z_scores = np.abs((data - self.mean) / self.std)
        return z_scores > self.threshold

Q12: Algorithm Challenge

Efficiently match meter readings to billing periods.

from bisect import bisect_right

def match_readings_to_periods(readings, billing_periods):
    period_starts = [p['start'] for p in billing_periods]
    matches = []
    
    for reading in readings:
        idx = bisect_right(period_starts, reading['date']) - 1
        if idx >= 0:
            matches.append({'reading': reading, 'period': billing_periods[idx]})
    
    return matches  # O(n log m) vs O(n*m) for naive approach
